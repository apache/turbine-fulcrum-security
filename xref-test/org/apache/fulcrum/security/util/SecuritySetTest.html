21  import junit.framework.TestCase;
22  
23  import org.apache.fulcrum.security.entity.Group;
24  import org.apache.fulcrum.security.model.dynamic.entity.impl.DynamicGroupImpl;
25  
26  /**
27   * Test using a SecuritySet. Useing various subclasses since it is Abstract.
28   *
29   * @author <a href="mailto:epugh@upstate.com">Eric Pugh</a>
30   * @version $Id$
31   */
32  public class SecuritySetTest extends TestCase
33  {
34  
35      /**
36       * Defines the testcase name for JUnit.
37       *
38       * @param name
39       *            the testcase's name.
40       */
41      public SecuritySetTest(String name)
42      {
43          super(name);
44      }
45  
46      public void testNull() throws Exception
47      {
48          SecuritySet<Group> securitySet = new GroupSet();
49          assertFalse(securitySet.contains(null));
50      }
51  
52      public void testContainsName()
53      {
54          SecuritySet<Group> securitySet = new GroupSet();
55          assertFalse(securitySet.containsName(null));
56          Group g = new DynamicGroupImpl();
57          g.setName("BOB");
58  
59          ((GroupSet) securitySet).add(g);
60          assertTrue(((GroupSet) securitySet).containsName("bob"));
61          assertTrue(((GroupSet) securitySet).containsName("BOB"));
62  
63      }
64  
65      public void testRemoveAll()
66      {
67          SecuritySet<Group> securitySet = new GroupSet();
68          assertFalse(securitySet.containsName(null));
69          Group g = new DynamicGroupImpl();
70          g.setName("BOB");
71          g.setId("BOB");
72  
73          ((GroupSet) securitySet).add(g);
74  
75          SecuritySet<Group> securitySet2 = new GroupSet();
76          assertFalse(securitySet.containsName(null));
77          g = new DynamicGroupImpl();
78          g.setName("BOB");
79          g.setId("BOB");
80  
81          ((GroupSet) securitySet2).add(g);
82          securitySet.removeAll(securitySet2);
83          assertEquals(0, securitySet.size());
84      }
85  
86      public void testToArray() throws Exception
87      {
88          SecuritySet<Group> securitySet = getTestData();
89          Object array[] = securitySet.toArray();
90          assertEquals(2, array.length);
91          Object array2[] = new Object[2];
92          array2[0] = "hi";
93          Object array3[] = securitySet.toArray(array2);
94          assertEquals(2, array3.length);
95      }
96  
97      public void testAdd() throws Exception
98      {
99          GroupSet securitySet = (GroupSet) getTestData();
100         Group g = new DynamicGroupImpl();
101         g.setName("Michael");
102         g.setId("Michael");
103         assertTrue(securitySet.add(g));
104     }
105 
106     private SecuritySet<Group> getTestData()
107     {
108         SecuritySet<Group> securitySet = new GroupSet();
109         assertFalse(securitySet.containsName(null));
110         Group g = new DynamicGroupImpl();
111         g.setName("JOE");
112         g.setId("JOE");
113 
114         Group g2 = new DynamicGroupImpl();
115         g2.setName("RICK");
116         g2.setId("RICK");
117 
118         ((GroupSet) securitySet).add(g);
119         ((GroupSet) securitySet).add(g2);
120 
121         return securitySet;
122     }
123 
124 }
