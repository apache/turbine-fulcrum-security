021import org.apache.avalon.framework.activity.Disposable;
022import org.apache.avalon.framework.activity.Initializable;
023import org.apache.avalon.framework.configuration.Configurable;
024import org.apache.avalon.framework.configuration.ConfigurationException;
025import org.apache.fulcrum.security.entity.SecurityEntity;
026import org.apache.fulcrum.security.spi.AbstractManager;
027import org.apache.fulcrum.security.util.DataBackendException;
028import org.apache.fulcrum.security.util.UnknownEntityException;
029import org.hibernate.HibernateException;
030import org.hibernate.Session;
031import org.hibernate.SessionFactory;
032import org.hibernate.Transaction;
033import org.hibernate.cfg.Configuration;
034
035/**
036 * 
037 * This persistenceHelper expects you to either pass in a SessionFactory to use,
038 * or it will create one from a hibernate.cfg.xml in the root of the classpath.
039 * 
040 * @author <a href="mailto:epugh@upstate.com">Eric Pugh</a>
041 * @version $Id: PersistenceHelperDefaultImpl.java 1374014 2012-08-16 19:47:27Z
042 *          tv $
043 */
044public class PersistenceHelperDefaultImpl extends AbstractManager implements PersistenceHelper, Configurable, Initializable, Disposable
045{
046    private Configuration configuration;
047    private SessionFactory sessionFactory;
048    private Session session;
049
050    /**
051     * Deletes an entity object
052     * 
053     * @param entity
054     *            The object to be removed
055     * @throws DataBackendException
056     *             if there was an error accessing the data backend.
057     */
058    public void removeEntity(SecurityEntity entity) throws DataBackendException
059    {
060        Transaction transaction = null;
061
062        try
063        {
064            Session session = retrieveSession();
065            transaction = session.beginTransaction();
066            session.delete(entity);
067            transaction.commit();
068            transaction = null;
069        }
070        catch (HibernateException he)
071        {
072            throw new DataBackendException("Problem removing entity:" + he.getMessage(), he);
073        }
074        finally
075        {
076            if (transaction != null)
077            {
078                transaction.rollback();
079            }
080        }
081    }
082
083    /**
084     * Stores changes made to an object
085     * 
086     * @param entity
087     *            The object to be saved
088     * @throws DataBackendException
089     *             if there was an error accessing the data backend.
090     */
091    public void updateEntity(SecurityEntity entity) throws DataBackendException
092    {
093        Transaction transaction = null;
094
095        try
096        {
097            Session session = retrieveSession();
098            transaction = session.beginTransaction();
099            session.update(entity);
100            transaction.commit();
101            transaction = null;
102        }
103        catch (HibernateException he)
104        {
105            if (he.getMessage().indexOf("Another object was associated with this id") > -1)
106            {
107                session.close();
108                updateEntity(entity);
109            }
110            else
111            {
112                throw new DataBackendException("updateEntity(" + entity + ")", he);
113            }
114        }
115        finally
116        {
117            if (transaction != null)
118            {
119                transaction.rollback();
120            }
121        }
122    }
123
124    /**
125     * adds an entity
126     * 
127     * @param entity
128     *            The object to be saved
129     * @throws DataBackendException
130     *             if there was an error accessing the data backend.
131     */
132    public void addEntity(SecurityEntity entity) throws DataBackendException
133    {
134        Transaction transaction = null;
135
136        try
137        {
138            Session session = retrieveSession();
139            transaction = session.beginTransaction();
140            session.save(entity);
141            transaction.commit();
142            transaction = null;
143        }
144        catch (HibernateException he)
145        {
146            throw new DataBackendException("addEntity(entity)", he);
147        }
148        finally
149        {
150            if (transaction != null)
151            {
152                transaction.rollback();
153            }
154        }
155    }
156
157    /**
158     * Returns a hibernate session, or if is null, opens one.
159     * 
160     * @return An Open hibernate session.
161     * @throws HibernateException generic exception
162     */
163    public Session retrieveSession() throws HibernateException
164    {
165        if (session == null)
166        {
167            session = getSessionFactory().openSession();
168        }
169        return session;
170    }
171
172    /**
173     * @return the hibernate configuration
174     */
175    public Configuration getConfiguration()
176    {
177        return configuration;
178    }
179
180    /**
181     * In some environments we will load the session factory up and pass it in.
182     * 
183     * @param sessionFactory
184     *            The hibernateService to set.
185     */
186    public void setSessionFactory(SessionFactory sessionFactory)
187    {
188        this.sessionFactory = sessionFactory;
189    }
190
191    /**
192     * Lazy loads the sessionFactory if it hasn't been requested yet.
193     * 
194     * @return the hibernate service
195     */
196    public SessionFactory getSessionFactory() throws HibernateException
197    {
198        return sessionFactory;
199    }
200
201    /* (non-Javadoc)
202     * @see org.apache.avalon.framework.activity.Initializable#initialize()
203     */
204    public void initialize() throws Exception
205    {
206        sessionFactory = configuration.buildSessionFactory();
207    }
208
209    /* (non-Javadoc)
210     * @see org.apache.fulcrum.security.spi.AbstractManager#dispose()
211     */
212    @Override
213    public void dispose()
214    {
215        sessionFactory.close();
216        super.dispose();
217    }
218
219    /* (non-Javadoc)
220     * @see org.apache.avalon.framework.configuration.Configurable#configure(org.apache.avalon.framework.configuration.Configuration)
221     */
222    public void configure(org.apache.avalon.framework.configuration.Configuration conf) throws ConfigurationException
223    {
224        configuration = new Configuration();
225
226        // read properties
227        org.apache.avalon.framework.configuration.Configuration[] props = conf.getChild("session-factory").getChildren("property");
228
229        for (org.apache.avalon.framework.configuration.Configuration prop : props)
230        {
231            String value = prop.getValue(null);
232            String key = prop.getAttribute("name", null);
233
234            if (key != null && value != null)
235            {
236                configuration.setProperty(key, value);
237            }
238            else
239            {
240                throw new ConfigurationException("Invalid configuration", prop);
241            }
242        }
243
244        // read mappings
245        org.apache.avalon.framework.configuration.Configuration[] maps = conf.getChild("mappings").getChildren("resource");
246
247        for (org.apache.avalon.framework.configuration.Configuration map : maps)
248        {
249            String value = map.getValue(null);
250
251            if (value != null)
252            {
253                configuration.addResource(value);
254            }
255            else
256            {
257                throw new ConfigurationException("Invalid configuration", map);
258            }
259        }
260    }
261}
