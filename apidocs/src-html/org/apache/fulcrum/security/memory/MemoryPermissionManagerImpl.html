021import java.util.ArrayList;
022import java.util.List;
023
024import org.apache.fulcrum.security.entity.Permission;
025import org.apache.fulcrum.security.spi.AbstractPermissionManager;
026import org.apache.fulcrum.security.util.DataBackendException;
027import org.apache.fulcrum.security.util.EntityExistsException;
028import org.apache.fulcrum.security.util.PermissionSet;
029import org.apache.fulcrum.security.util.UnknownEntityException;
030
031/**
032 * This implementation keeps all objects in memory. This is mostly meant to help
033 * with testing and prototyping of ideas.
034 *
035 * @author <a href="mailto:epugh@upstate.com">Eric Pugh</a>
036 * @version $Id: MemoryPermissionManagerImpl.java 1374008 2012-08-16 19:42:18Z
037 *          tv $
038 */
039public class MemoryPermissionManagerImpl extends AbstractPermissionManager
040{
041    // static to simulate database ?
042    private static List<Permission> permissions = new ArrayList<Permission>();
043    
044    public MemoryPermissionManagerImpl() {
045        permissions.clear();
046    }
047
048    /** Our Unique ID counter */
049    // private static int uniqueId = 0;
050
051    /**
052     * Retrieves all permissions defined in the system.
053     *
054     * @return the names of all permissions defined in the system.
055     * @throws DataBackendException
056     *             if there was an error accessing the data backend.
057     */
058    public PermissionSet getAllPermissions() throws DataBackendException
059    {
060        return new PermissionSet(permissions);
061    }
062
063    /**
064     * Renames an existing Permission.
065     *
066     * @param permission
067     *            The object describing the permission to be renamed.
068     * @param name
069     *            the new name for the permission.
070     * @throws DataBackendException
071     *             if there was an error accessing the data backend.
072     * @throws UnknownEntityException
073     *             if the permission does not exist.
074     */
075    public synchronized void renamePermission(Permission permission, String name) throws DataBackendException, UnknownEntityException
076    {
077        boolean permissionExists = false;
078        try
079        {
080            permissionExists = checkExists(permission);
081            if (permissionExists)
082            {
083                permissions.remove(permission);
084                permission.setName(name);
085                permissions.add(permission);
086                return;
087            }
088        }
089        catch (DataBackendException e)
090        {
091            throw new DataBackendException("renamePermission(Permission,name)", e);
092        }
093
094        throw new UnknownEntityException("Unknown permission '" + permission + "'");
095    }
096
097    /**
098     * Determines if the <code>Permission</code> exists in the security system.
099     *
100     * @param permissionName
101     *            a <code>String</code> value
102     * @return <code>true</code> if the permission exists in the system, false otherwise
103     * @throws DataBackendException
104     *             when more than one Permission with the same name exists.
105     */
106    public boolean checkExists(String permissionName) throws DataBackendException
107    {
108        return MemoryHelper.checkExists(permissions, permissionName);
109    }
110
111    /**
112     * Removes a Permission from the system.
113     *
114     * @param permission
115     *            The object describing the permission to be removed.
116     * @throws DataBackendException
117     *             if there was an error accessing the data backend.
118     * @throws UnknownEntityException
119     *             if the permission does not exist.
120     */
121    public synchronized void removePermission(Permission permission) throws DataBackendException, UnknownEntityException
122    {
123        boolean permissionExists = false;
124        try
125        {
126            permissionExists = checkExists(permission);
127            if (permissionExists)
128            {
129                permissions.remove(permission);
130                return;
131            }
132        }
133        catch (DataBackendException e)
134        {
135            throw new DataBackendException("removePermission(Permission)", e);
136        }
137
138        throw new UnknownEntityException("Unknown permission '" + permission + "'");
139    }
140
141    /**
142     * Creates a new permission with specified attributes.
143     *
144     * @param permission
145     *            the object describing the permission to be created.
146     * @return a new Permission object that has id set up properly.
147     * @throws DataBackendException
148     *             if there was an error accessing the data backend.
149     */
150    @Override
151    protected synchronized <T extends Permission> T persistNewPermission(T permission) throws DataBackendException
152    {
153        permission.setId(MemoryHelper.getUniqueId());
154        permissions.add(permission);
155        return permission;
156    }
157}
