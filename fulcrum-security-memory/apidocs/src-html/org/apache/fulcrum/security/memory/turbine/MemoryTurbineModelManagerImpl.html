021import org.apache.fulcrum.security.entity.Group;
022import org.apache.fulcrum.security.entity.Permission;
023import org.apache.fulcrum.security.entity.Role;
024import org.apache.fulcrum.security.entity.User;

026import org.apache.fulcrum.security.model.turbine.TurbineModelManager;
027import org.apache.fulcrum.security.model.turbine.entity.TurbineGroup;
028import org.apache.fulcrum.security.model.turbine.entity.TurbinePermission;
029import org.apache.fulcrum.security.model.turbine.entity.TurbineRole;
030import org.apache.fulcrum.security.model.turbine.entity.TurbineUser;
031import org.apache.fulcrum.security.model.turbine.entity.TurbineUserGroupRole;
032import org.apache.fulcrum.security.util.DataBackendException;
033import org.apache.fulcrum.security.util.UnknownEntityException;

035/**
036 * This implementation keeps all objects in memory. This is mostly meant to help
037 * with testing and prototyping of ideas.
038 * 
039 * @author <a href="mailto:epugh@upstate.com">Eric Pugh </a>
040 * @version $Id: MemoryTurbineModelManagerImpl.java,v 1.2 2004/07/07 16:51:27
041 *          epugh Exp $
042 */
043public class MemoryTurbineModelManagerImpl extends AbstractTurbineModelManager implements TurbineModelManager
044{
045    /**
046     * Grants a Role a Permission
047     * 
048     * @param role
049     *            the Role.
050     * @param permission
051     *            the Permission.
052     * @throws DataBackendException
053     *             if there was an error accessing the data backend.
054     * @throws UnknownEntityException
055     *             if role or permission is not present.
056     */
057    public synchronized void grant(Role role, Permission permission) throws DataBackendException, UnknownEntityException
058    {
059        boolean roleExists = false;
060        boolean permissionExists = false;
061        try
062        {
063            roleExists = getRoleManager().checkExists(role);
064            permissionExists = getPermissionManager().checkExists(permission);
065            if (roleExists && permissionExists)
066            {
067                ((TurbineRole) role).addPermission(permission);
068                ((TurbinePermission) permission).addRole(role);
069                return;
070            }
071        }
072        catch (Exception e)
073        {
074            throw new DataBackendException("grant(Role,Permission) failed", e);
075        }
076
077        if (!roleExists)
078        {
079            throw new UnknownEntityException("Unknown role '" + role.getName() + "'");
080        }
081        if (!permissionExists)
082        {
083            throw new UnknownEntityException("Unknown permission '" + permission.getName() + "'");
084        }
085    }
086
087    /**
088     * Revokes a Permission from a Role.
089     * 
090     * @param role
091     *            the Role.
092     * @param permission
093     *            the Permission.
094     * @throws DataBackendException
095     *             if there was an error accessing the data backend.
096     * @throws UnknownEntityException
097     *             if role or permission is not present.
098     */
099    public synchronized void revoke(Role role, Permission permission) throws DataBackendException, UnknownEntityException
100    {
101        boolean roleExists = false;
102        boolean permissionExists = false;
103        try
104        {
105            roleExists = getRoleManager().checkExists(role);
106            permissionExists = getPermissionManager().checkExists(permission);
107            if (roleExists && permissionExists)
108            {
109                ((TurbineRole) role).removePermission(permission);
110                ((TurbinePermission) permission).removeRole(role);
111                return;
112            }
113        }
114        catch (Exception e)
115        {
116            throw new DataBackendException("revoke(Role,Permission) failed", e);
117        }
118
119        if (!roleExists)
120        {
121            throw new UnknownEntityException("Unknown role '" + role.getName() + "'");
122        }
123        if (!permissionExists)
124        {
125            throw new UnknownEntityException("Unknown permission '" + permission.getName() + "'");
126        }
127    }
128
129    public void grant(User user, Group group, Role role) throws DataBackendException, UnknownEntityException
130    {
131        boolean roleExists = false;
132        boolean userExists = false;
133        boolean groupExists = false;
134        try
135        {
136            roleExists = getRoleManager().checkExists(role);
137            userExists = getUserManager().checkExists(user);
138            groupExists = getGroupManager().checkExists(group);
139            if (roleExists && groupExists && userExists)
140            {
141                TurbineUserGroupRole ugr = new TurbineUserGroupRole();
142                ugr.setGroup(group);
143                ugr.setRole(role);
144                ugr.setUser(user);
145                ((TurbineUser) user).addUserGroupRole(ugr);
146                ((TurbineGroup) group).addUserGroupRole(ugr);
147                ((TurbineRole) role).addUserGroupRole(ugr);
148                return;
149            }
150        }
151        catch (Exception e)
152        {
153            throw new DataBackendException("grant(Role,Permission) failed", e);
154        }
155
156        if (!roleExists)
157        {
158            throw new UnknownEntityException("Unknown role '" + role.getName() + "'");
159        }
160        if (!groupExists)
161        {
162            throw new UnknownEntityException("Unknown group '" + group.getName() + "'");
163        }
164        if (!userExists)
165        {
166            throw new UnknownEntityException("Unknown user '" + user.getName() + "'");
167        }
168
169    }
170
171    public void revoke(User user, Group group, Role role) throws DataBackendException, UnknownEntityException
172    {
173        boolean roleExists = false;
174        boolean userExists = false;
175        boolean groupExists = false;
176        try
177        {
178            roleExists = getRoleManager().checkExists(role);
179            userExists = getUserManager().checkExists(user);
180            groupExists = getGroupManager().checkExists(group);
181            if (roleExists && groupExists && userExists)
182            {
183                boolean ugrFound = false;
184                for (TurbineUserGroupRole ugr : ((TurbineUser) user).getUserGroupRoleSet())
185                {
186                    if (ugr.getUser().equals(user) && ugr.getGroup().equals(group) && ugr.getRole().equals(role))
187                    {
188                        ugrFound = true;
189
190                        ((TurbineUser) user).removeUserGroupRole(ugr);
191                        ((TurbineGroup) group).removeUserGroupRole(ugr);
192                        ((TurbineRole) role).removeUserGroupRole(ugr);
193
194                        break;
195                    }
196                }
197                if (!ugrFound)
198                {
199                    throw new UnknownEntityException("Could not find User/Group/Role");
200                }
201
202                return;
203            }
204        }
205        catch (Exception e)
206        {
207            throw new DataBackendException("grant(Role,Permission) failed", e);
208        }
209
210        if (!roleExists)
211        {
212            throw new UnknownEntityException("Unknown role '" + role.getName() + "'");
213        }
214        if (!groupExists)
215        {
216            throw new UnknownEntityException("Unknown group '" + group.getName() + "'");
217        }
218        if (!userExists)
219        {
220            throw new UnknownEntityException("Unknown user '" + user.getName() + "'");
221        }
222
223    }
224
225    @Override
226    public void replace( User user, Role oldRole, Role newRole )
227        throws DataBackendException, UnknownEntityException
228    {
229        Group group = getGlobalGroup();
230        revoke( user, group, oldRole );
231        grant( user, group, newRole );
232    }
233}
