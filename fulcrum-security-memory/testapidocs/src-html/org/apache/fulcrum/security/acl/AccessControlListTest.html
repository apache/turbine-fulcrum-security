086    @Test
087    public void testCreatingDefaultAccessControlListViaFactory() throws Exception
088    {
089        Group group = getGroup();
090        Role role = getRole();
091        Permission permission = getPermission();
092        user = userManager.getUserInstance("User 1");
093        userManager.addUser(user, "secretpassword");
094        modelManager.grant(user, group);
095        modelManager.grant(group, role);
096        modelManager.grant(role, permission);
097        RoleSet roleSet = new RoleSet();
098        PermissionSet permissionSet = new PermissionSet();
099        roleSet.add(role);
100        permissionSet.add(permission);
101        Map<Group, RoleSet> roleSets = new HashMap<Group, RoleSet>();
102        Map<Role, PermissionSet> permissionSets = new HashMap<Role, PermissionSet>();
103        roleSets.put(group, roleSet);
104        permissionSets.put(role, permissionSet);
105        AccessControlList acl = userManager.getACL(user);
106        assertTrue(acl instanceof DynamicAccessControlList);
107        DynamicAccessControlList dacl = (DynamicAccessControlList) acl;
108        assertTrue(dacl.hasRole(role));
109        assertTrue(dacl.hasPermission(permission));
110        
111        Serializable serDeSer = SerializationUtils.roundtrip(acl);
112        assertTrue( ((DynamicAccessControlList)serDeSer).getRoles().toString().equals(dacl.getRoles().toString()));
113
114    }
