101    @Test
102    public void testCreatingDefaultAccessControlListViaFactory() throws Exception
103    {
104        Group group = getGroup();
105        Role role = getRole();
106        Role globalRole = getRole();
107        Permission permission = getPermission();
108        modelManager.grant(user, modelManager.getGlobalGroup(), globalRole);
109        modelManager.grant(user, group, role);
110        modelManager.grant(role, permission);
111        RoleSet roleSet = new RoleSet();
112        PermissionSet permissionSet = new PermissionSet();
113        roleSet.add(role);
114        permissionSet.add(permission);
115        Map<Group, RoleSet> roleSets = new HashMap<Group, RoleSet>();
116        Map<Group, PermissionSet> permissionSets = new HashMap<Group, PermissionSet>();
117        roleSets.put(group, roleSet);
118        permissionSets.put(group, permissionSet);
119        AccessControlList acl = userManager.getACL(user);
120        assertTrue(acl instanceof TurbineAccessControlList);
121        TurbineAccessControlList dacl = (TurbineAccessControlList) acl;
122        assertTrue(dacl.hasRole(role,group));
123        assertTrue(dacl.hasRole(globalRole));
124        assertTrue(dacl.hasPermission(permission,group));     
125        assertEquals(dacl.getPermissions(group).toString(),permissionSets.get(group).toString());
126        Serializable serDeSer = SerializationUtils.roundtrip(acl);
127        assertTrue( ((TurbineAccessControlList)serDeSer).getRoles().toString().equals(dacl.getRoles().toString()),
128                "Expected RoleSet: [role 4 -> 4]");
129
130    }
