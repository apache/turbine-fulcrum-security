326    @Test
327    public void testAddRemoveDelegate() throws Exception
328    {
329        DynamicUser borris = (DynamicUser) userManager.getUserInstance(USERNAME_BORRIS);
330        userManager.addUser(borris, "mc");
331        DynamicUser sam = (DynamicUser) userManager.getUserInstance(USERNAME_SAM);
332        userManager.addUser(sam, "mc");
333        modelManager.addDelegate(borris, sam);
334        assertTrue(borris.getDelegatees().contains(sam));
335        assertTrue(sam.getDelegators().contains(borris));
336
337        DynamicUser borrisLoaded = (DynamicUser) userManager.getUser(USERNAME_BORRIS);
338        DynamicUser samLoaded = (DynamicUser) userManager.getUser(USERNAME_SAM);
339        assertTrue(borrisLoaded.getDelegatees().contains(samLoaded));
340        assertTrue(samLoaded.getDelegators().contains(borrisLoaded));
341
342        // Now grant borris some permissions and check sam has them
343        Group group = groupManager.getGroupInstance();
344        group.setName(ONLY_BORRIS_GROUP);
345        groupManager.addGroup(group);
346        Role role = roleManager.getRoleInstance();
347        role.setName(ONLY_BORRIS_ROLE);
348        roleManager.addRole(role);
349        Permission permission = permissionManager.getPermissionInstance();
350        permission.setName(ONLY_BORRIS_PERMISSION);
351        permissionManager.addPermission(permission);
352
353        modelManager.grant(role, permission);
354        modelManager.grant(group, role);
355        modelManager.grant(borris, group);
356
357        DynamicAccessControlList acl = (DynamicAccessControlList) userManager.getACL(sam);
358        assertTrue(acl.hasPermission(permission));
359        assertTrue(acl.hasRole(role));
360
361        // Now just to be silly make it recursive and check permissions work
362        modelManager.addDelegate(sam, borris);
363        acl = (DynamicAccessControlList) userManager.getACL(sam);
364        assertTrue(acl.hasPermission(permission));
365        assertTrue(acl.hasRole(role));
366
367        modelManager.removeDelegate(borris, sam);
368        assertFalse(borris.getDelegatees().contains(sam));
369        assertFalse(sam.getDelegators().contains(borris));
370
371        borrisLoaded = (DynamicUser) userManager.getUser(USERNAME_BORRIS);
372        samLoaded = (DynamicUser) userManager.getUser(USERNAME_SAM);
373        assertFalse(borrisLoaded.getDelegatees().contains(samLoaded));
374        assertFalse(samLoaded.getDelegators().contains(borrisLoaded));
375
376        boolean thrown = false;
377        try
378        {
379            modelManager.removeDelegate(borris, sam);
380        }
381        catch (DataBackendException e)
382        {
383            throw e;
384        }
385        catch (UnknownEntityException e)
386        {
387            thrown = true;
388        }
389        assertTrue(thrown);
390    }
