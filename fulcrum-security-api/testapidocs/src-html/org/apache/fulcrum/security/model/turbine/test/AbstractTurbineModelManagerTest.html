174    {
175        Group group = securityService.getGroupManager().getGroupInstance();
176        group.setName("TEST_REVOKEALLUSER_GROUP");
177        securityService.getGroupManager().addGroup(group);
178        Role role = securityService.getRoleManager().getRoleInstance();
179        role.setName("TEST_REVOKEALLUSER_ROLE");
180        //role.setId(new Integer(1));
181        role = securityService.getRoleManager().addRole(role);
182        
183        Group group2 = securityService.getGroupManager().getGroupInstance();
184        group2.setName("TEST_REVOKEALLUSER_GROUP2");
185        //group2.setId(new Integer(1));
186        securityService.getGroupManager().addGroup(group2);
187        Role role2 = securityService.getRoleManager().getRoleInstance();
188        role2.setName("TEST_REVOKEALLUSER_ROLE2");
189        //role2.setId(new Integer(2));
190        role2 = securityService.getRoleManager().addRole(role2);
191        
192        String username = "calvin";
193        User user = userManager.getUserInstance(username);
194        user = userManager.addUser(user, username);
195        modelManager.grant(user, group, role);
196        modelManager.grant(user, group2, role2);
197        // original objects have relationship attached
198        Set<TurbineUserGroupRole> userGroupRoleSet =  ((TurbineUser)user).getUserGroupRoleSet();
199        assertEquals(2, userGroupRoleSet.size());
200        Set<TurbineUserGroupRole> userGroupRoleSet1 = ((TurbineRole) role).getUserGroupRoleSet();
201        assertEquals(1, userGroupRoleSet1.size());
202        Set<TurbineUserGroupRole> userGroupRoleSet2 = ((TurbineGroup) group).getUserGroupRoleSet();
203        assertEquals(1, userGroupRoleSet2.size());
204        // retrieve objects again, which have as lazily loaded no usergroupset yet
205        group = groupManager.getGroupById(group.getId());
206        user = userManager.getUser( username );
207        role = roleManager.getRoleById( role.getId() );
208        
209        UserSet<User> userSet = userManager.getAllUsers();
210        User user2 = userManager.getUserById( user.getId() );
211        assertEquals( user, user2 );
212        
213        // retrieve usergroupset now
214        userGroupRoleSet =  ((TurbineUser)user).getUserGroupRoleSet();
215        assertEquals(2, userGroupRoleSet.size());
216        userGroupRoleSet1 = ((TurbineRole) role).getUserGroupRoleSet();
217        assertEquals(1, userGroupRoleSet1.size());
218        userGroupRoleSet2 = ((TurbineGroup) group).getUserGroupRoleSet();
219        assertEquals(1, userGroupRoleSet2.size());
220
221        modelManager.revokeAll(user);
222        group = groupManager.getGroupById(group.getId());
223        assertEquals(0, ((TurbineGroup) group).getUserGroupRoleSet().size());
224        role = securityService.getRoleManager().getRoleByName("TEST_REVOKEALLUSER_ROLE");
225        assertEquals(0,((TurbineRole) role).getUserGroupRoleSet().size());
226        assertTrue(((TurbineRole) role).getUserGroupRoleSet().isEmpty());
227        
228        modelManager.grant(user, group, role);
229        assertEquals(1,((TurbineRole) role).getUserGroupRoleSet().size());
230        assertTrue(((TurbineRole) role).getUserGroupRoleSet().iterator().next().getGroup().equals( group ));

234    @Test
235    public void testGrantUserGroupRole() throws Exception
236    {
237        Group group = securityService.getGroupManager().getGroupInstance();
238        group.setName("TEST_GROUP");
239        securityService.getGroupManager().addGroup(group);
240        Role role = roleManager.getRoleInstance();
241        role.setName("TEST_Role");
242        roleManager.addRole(role);
243        User user = userManager.getUserInstance("Clint");
244        userManager.addUser(user, "clint");
245        modelManager.grant(user, group, role);
246        boolean ugrFound = false;
247        TurbineUserGroupRole ugrTest = null;
248        for (TurbineUserGroupRole ugr : ((TurbineUser) user).getUserGroupRoleSet())
249        {
250            if (ugr.getUser().equals(user) && ugr.getGroup().equals(group) && ugr.getRole().equals(role))
251            {
252                ugrFound = true;
253                ugrTest = ugr;
254                break;
255            }
256        }
257        assertTrue(ugrFound);
258        assertTrue(ugrTest.getGroup().equals(group));
259        assertTrue(ugrTest.getUser().equals(user));

288    {
289        Permission permission = permissionManager.getPermissionInstance();
290        permission.setName("ANSWER_PHONE__");
291        //permission.setId(new Integer (1));
292        permissionManager.addPermission(permission);
293        
294        Permission permission2 = permissionManager.getPermissionInstance();
295        permission2.setName("ANSWER_PHONE__2");
296        // TurbineHibernate.hbm.xml has <generator class="native" for id, throws org.hibernate.HibernateException: 
297        // identifier of an instance of org.apache.fulcrum.security.model.turbine.entity.impl.TurbinePermissionImpl was altered from 1 to 2
298        //permission.setId(new Integer (2));
299        permissionManager.addPermission(permission2);
300        
301        role = roleManager.getRoleInstance("RECEPTIONIST__");
302        roleManager.addRole(role);
303        
304        Group group = securityService.getGroupManager().getGroupInstance();
305        group.setName("TEST_GROUP__");
306        //group.setId(new Integer(1));
307        securityService.getGroupManager().addGroup(group);
308        Role role = roleManager.getRoleInstance();
309        
310        role.setName("TEST_Role__");
311        //role.setId(new Integer(1));
312        roleManager.addRole(role);
313        User user = userManager.getUserInstance("Clint__");
314        userManager.addUser(user, "clint");
315        // this is required
316        modelManager.grant(user, group, role);
317        
318        modelManager.grant(role, permission);
319        modelManager.grant(role, permission2);
320        role = roleManager.getRoleById(role.getId());
321        PermissionSet permissions = ((TurbineRole) role).getPermissions();
322        assertEquals(2, permissions.size());
323        assertTrue(((TurbineRole) role).getPermissions().contains(permission));
324        assertTrue(((TurbineRole) role).getPermissions().contains(permission2));
325        
326        modelManager.revoke( role, permission2 );
327        permissions = ((TurbineRole) role).getPermissions();
328        assertEquals(1, permissions.size());
329        assertTrue(((TurbineRole) role).getPermissions().contains(permission));
330        
331        modelManager.revoke(role, permission);
332        permissions = ((TurbineRole) role).getPermissions();
333        assertEquals(0, permissions.size());
334        assertFalse(((TurbineRole) role).getPermissions().contains(permission));

339    @Test
340    public void testReplaceUserGroupRole() throws Exception
341    {
342        Group global = modelManager.getGlobalGroup();
343        
344        Role role = roleManager.getRoleInstance();
345        role.setName("TEST_REPLACE_ROLE");
346        if (!roleManager.checkExists( role ) ) {
347            roleManager.addRole(role);
348        }
349
350        
351        
352        Role newRole = roleManager.getRoleInstance();
353        newRole.setName("TEST_NEW_ROLE");
354        roleManager.addRole(newRole);
355        User user = userManager.getUserInstance("Dave");
356        userManager.addUser(user, "dave");
357        
358        modelManager.grant(user, global, role);
359        
360        modelManager.replace(user, role, newRole);
361        
362
363        
364        boolean ugrFound = false;
365        boolean ugrNotFound = true;
366        TurbineUserGroupRole ugrTest = null;
367        for (TurbineUserGroupRole ugr : ((TurbineUser) user).getUserGroupRoleSet())
368        {
369            if (ugr.getUser().equals(user) && ugr.getGroup().equals(global) && ugr.getRole().equals(newRole))
370            {
371                ugrFound = true;
372                ugrTest = ugr;
373            }
374            if (ugr.getUser().equals(user) && ugr.getGroup().equals(global) && ugr.getRole().equals(role))
375            {
376                ugrNotFound = false;
377                ugrTest = ugr;
378            }
379        }
380        assertTrue(ugrFound);
381        assertTrue(ugrNotFound);
382        assertTrue(ugrTest.getGroup().equals(global));
383        assertTrue(ugrTest.getUser().equals(user));
384
385    }
