029import org.apache.fulcrum.security.GroupManager;
030import org.apache.fulcrum.security.SecurityService;
031import org.apache.fulcrum.security.entity.Group;
032import org.apache.fulcrum.security.model.dynamic.entity.DynamicGroup;
033import org.apache.fulcrum.security.util.EntityExistsException;
034import org.apache.fulcrum.security.util.GroupSet;
035import org.apache.fulcrum.security.util.UnknownEntityException;
036import org.apache.fulcrum.testcontainer.BaseUnit5Test;
037import org.junit.jupiter.api.Test;
038
039/**
040 * @author Eric Pugh
041 */
042public abstract class AbstractGroupManagerTest extends BaseUnit5Test
043{
044    protected Group group;
045    protected GroupManager groupManager;
046    protected SecurityService securityService;
047
048
049    /*
050     * Class to test for Group getGroupInstance()
051     */
052    @Test
053    public void testGetGroupInstance() throws Exception
054    {
055        group = groupManager.getGroupInstance();
056        assertNotNull(group);
057        assertTrue(group.getName() == null);
058    }
059
060    /*
061     * Class to test for Group getGroupInstance(String)
062     */
063    @Test
064    public void testGetGroupInstanceString() throws Exception
065    {
066        group = groupManager.getGroupInstance("DOG_CATCHER");
067        assertEquals("DOG_CATCHER".toLowerCase(), group.getName());
068    }
069    @Test
070    public void testGetGroup() throws Exception
071    {
072        group = groupManager.getGroupInstance("DOG_CATCHER2");
073        groupManager.addGroup(group);
074        Group group2 = groupManager.getGroupByName("DOG_CATCHER2");
075        assertEquals(group.getName(), group2.getName());
076    }
077    @Test
078    public void testGetGroupByName() throws Exception
079    {
080        group = groupManager.getGroupInstance("CLEAN_KENNEL");
081        groupManager.addGroup(group);
082        Group group2 = groupManager.getGroupByName("CLEAN_KENNEL");
083        assertEquals(group.getName(), group2.getName());
084        group2 = groupManager.getGroupByName("Clean_KeNNel");
085        assertEquals(group.getName(), group2.getName());
086    }
087    @Test
088    public void testGetGroupById() throws Exception
089    {
090        group = groupManager.getGroupInstance("CLEAN_KENNEL_A");
091        groupManager.addGroup(group);
092        Group group2 = groupManager.getGroupById(group.getId());
093        assertEquals(group.getName(), group2.getName());
094    }
095    @Test
096    public void testGetAllGroups() throws Exception
097    {
098        int size = groupManager.getAllGroups().size();
099        group = groupManager.getGroupInstance("CLEAN_KENNEL_J");
100        groupManager.addGroup(group);
101        GroupSet groupSet = groupManager.getAllGroups();
102        assertEquals(size + 1, groupSet.size());
103    }
104    @Test
105    public void testRemoveGroup() throws Exception
106    {
107        group = groupManager.getGroupInstance("CLEAN_KENNEL_K");
108        groupManager.addGroup(group);
109        int size = groupManager.getAllGroups().size();
110        if (group instanceof DynamicGroup)
111        {
112            assertEquals(0, ((DynamicGroup) group).getUsers().size());
113            assertEquals(0, ((DynamicGroup) group).getRoles().size());
114        }
115        groupManager.removeGroup(group);
116        try
117        {
118            Group group2 = groupManager.getGroupById(group.getId());
119            fail("Should have thrown UEE");
120        }
121        catch (UnknownEntityException uee)
122        {
123            // good
124        }
125        assertEquals(size - 1, groupManager.getAllGroups().size());
126    }
127    @Test
128    public void testRenameGroup() throws Exception
129    {
130        group = groupManager.getGroupInstance("CLEAN_KENNEL_X");
131        groupManager.addGroup(group);
132        int size = groupManager.getAllGroups().size();
133        groupManager.renameGroup(group, "CLEAN_GROOMING_ROOM");
134        Group group2 = groupManager.getGroupById(group.getId());
135        assertEquals("CLEAN_GROOMING_ROOM".toLowerCase(), group2.getName());
136        assertEquals(size, groupManager.getAllGroups().size());
137    }
138    @Test
139    public void testCheckExists() throws Exception
140    {
141        group = groupManager.getGroupInstance("GREET_PEOPLE");
142        groupManager.addGroup(group);
143        assertTrue(groupManager.checkExists(group));
144        Group group2 = groupManager.getGroupInstance("WALK_DOGS");
145        assertFalse(groupManager.checkExists(group2));
146    }
147    @Test
148    public void testCheckExistsWithString() throws Exception
149    {
150        group = groupManager.getGroupInstance("GREET_PEOPLE2");
151        groupManager.addGroup(group);
152        assertTrue(groupManager.checkExists(group.getName()));
153        Group group2 = groupManager.getGroupInstance("WALK_DOGS2");
154        assertFalse(groupManager.checkExists(group2.getName()));
155    }
156
157    /*
158     * Class to test for boolean checkExists(string)
159     */
160    @Test
161    public void testAddGroupTwiceFails() throws Exception
162    {
163        group = groupManager.getGroupInstance("EATLUNCH");
164        groupManager.addGroup(group);
165        assertTrue(groupManager.checkExists(group.getName()));
166        Group group2 = groupManager.getGroupInstance("EATLUNCH");
167        try
168        {
169            groupManager.addGroup(group2);
170        }
171        catch (EntityExistsException uee)
172        {
173            // good
174        }
175    }
176    @Test
177    public void testAddGroup() throws Exception
178    {
179        group = groupManager.getGroupInstance("CLEAN_RABBIT_HUTCHES");
180        assertNull(group.getId());
181        groupManager.addGroup(group);
182        assertNotNull(group.getId());
183        assertNotNull(groupManager.getGroupById(group.getId()));
184    }
185
186}
