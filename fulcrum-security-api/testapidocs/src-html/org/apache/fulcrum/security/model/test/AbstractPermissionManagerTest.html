022import org.apache.fulcrum.security.PermissionManager;
023import org.apache.fulcrum.security.SecurityService;
024import org.apache.fulcrum.security.entity.Permission;
025import org.apache.fulcrum.security.entity.Role;
026import org.apache.fulcrum.security.model.dynamic.DynamicModelManager;
027import org.apache.fulcrum.security.model.dynamic.entity.DynamicRole;
028import org.apache.fulcrum.security.util.EntityExistsException;
029import org.apache.fulcrum.security.util.PermissionSet;
030import org.apache.fulcrum.security.util.UnknownEntityException;
031import org.apache.fulcrum.testcontainer.BaseUnit5Test;
032import org.junit.jupiter.api.Test;
033
034import static org.junit.jupiter.api.Assertions.*;
035
036/**
037 * @author Eric Pugh
038 */
039public abstract class AbstractPermissionManagerTest extends BaseUnit5Test
040{
041    protected Permission permission;
042    protected PermissionManager permissionManager;
043    protected SecurityService securityService;
044
045    /*
046     * Class to test for Permission getPermissionInstance()
047     */
048    @Test
049    public void testGetPermissionInstance() throws Exception
050    {
051        permission = permissionManager.getPermissionInstance();
052        assertNotNull(permission);
053        assertTrue(permission.getName() == null);
054    }
055
056    /*
057     * Class to test for Permission getPermissionInstance(String)
058     */
059    @Test
060    public void testGetPermissionInstanceString() throws Exception
061    {
062        permission = permissionManager.getPermissionInstance("CAN_TREAT_ANIMALS");
063        assertEquals("can_treat_animals", permission.getName());
064    }
065    @Test
066    public void testGetPermissionByName() throws Exception
067    {
068        permission = permissionManager.getPermissionInstance("CLEAN_KENNEL");
069        permissionManager.addPermission(permission);
070        Permission permission2 = permissionManager.getPermissionByName("CLEAN_KENNEL");
071        assertEquals(permission.getName(), permission2.getName());
072    }
073    @Test
074    public void testGetPermissionById() throws Exception
075    {
076        permission = permissionManager.getPermissionInstance("ADMINSTER_DRUGS");
077        permissionManager.addPermission(permission);
078        Permission permission2 = permissionManager.getPermissionById(permission.getId());
079        assertEquals(permission.getName(), permission2.getName());
080    }
081    @Test
082    public void testGetAllPermissions() throws Exception
083    {
084        int size = permissionManager.getAllPermissions().size();
085        permission = permissionManager.getPermissionInstance("WALK_DOGS");
086        permissionManager.addPermission(permission);
087        PermissionSet permissionSet = permissionManager.getAllPermissions();
088        assertEquals(size + 1, permissionSet.size());
089    }
090    @Test
091    public void testRenamePermission() throws Exception
092    {
093        permission = permissionManager.getPermissionInstance("CLEAN_FRONT_OFFICE");
094        permissionManager.addPermission(permission);
095        int size = permissionManager.getAllPermissions().size();
096        permissionManager.renamePermission(permission, "CLEAN_GROOMING_ROOM");
097        Permission permission2 = permissionManager.getPermissionById(permission.getId());
098        assertEquals("CLEAN_GROOMING_ROOM".toLowerCase(), permission2.getName());
099        assertEquals(size, permissionManager.getAllPermissions().size());
100    }
101    @Test
102    public void testRemovePermission() throws Exception
103    {
104        permission = permissionManager.getPermissionInstance("CLEAN_CAT_HOUSE");
105        permissionManager.addPermission(permission);
106        permissionManager.removePermission(permission);
107        try
108        {
109            permission = permissionManager.getPermissionById(permission.getId());
110            fail("Should have thrown UnknownEntityException");
111        }
112        catch (UnknownEntityException uee)
113        {
114            // good
115        }
116    }
117    @Test
118    public void testAddPermission() throws Exception
119    {
120        permission = permissionManager.getPermissionInstance("CLEAN_BIG_KENNEL");
121        assertNull(permission.getId());
122        permissionManager.addPermission(permission);
123        assertNotNull(permission.getId());
124        permission = permissionManager.getPermissionById(permission.getId());
125        assertNotNull(permission);
126    }
127
128    /*
129     * Class to test for PermissionSet getPermissions(Role)
130     */
131    @Test
132    public void testGetPermissionsRole() throws Exception
133    {
134        permission = permissionManager.getPermissionInstance("GREET_PEOPLE");
135        permissionManager.addPermission(permission);
136        Permission permission2 = permissionManager.getPermissionInstance("ADMINISTER_DRUGS");
137        permissionManager.addPermission(permission2);
138        Role role = securityService.getRoleManager().getRoleInstance("VET_TECH");
139        securityService.getRoleManager().addRole(role);
140        ((DynamicModelManager) securityService.getModelManager()).grant(role, permission);
141        PermissionSet permissions = ((DynamicRole) role).getPermissions();
142        assertEquals(1, permissions.size());
143        assertTrue(permissions.contains(permission));
144        assertFalse(permissions.contains(permission2));
145    }
146
147    /*
148     * Class to test for boolean checkExists(permission)
149     */
150    @Test
151    public void testCheckExistsPermission() throws Exception
152    {
153        permission = permissionManager.getPermissionInstance("OPEN_OFFICE");
154        permissionManager.addPermission(permission);
155        assertTrue(permissionManager.checkExists(permission));
156        Permission permission2 = permissionManager.getPermissionInstance("CLOSE_OFFICE");
157        assertFalse(permissionManager.checkExists(permission2));
158    }
159
160    /*
161     * Class to test for boolean checkExists(string)
162     */
163    @Test
164    public void testCheckExistsPermissionWithString() throws Exception
165    {
166        permission = permissionManager.getPermissionInstance("OPEN_OFFICE2");
167        permissionManager.addPermission(permission);
168        assertTrue(permissionManager.checkExists(permission.getName()));
169        Permission permission2 = permissionManager.getPermissionInstance("CLOSE_OFFICE2");
170        assertFalse(permissionManager.checkExists(permission2.getName()));
171    }
172
173    /*
174     * Class to test for boolean checkExists(string)
175     */
176    @Test
177    public void testAddPermissionTwiceFails() throws Exception
178    {
179        permission = permissionManager.getPermissionInstance("EATLUNCH");
180        permissionManager.addPermission(permission);
181        assertTrue(permissionManager.checkExists(permission.getName()));
182        Permission permission2 = permissionManager.getPermissionInstance("EATLUNCH");
183        try
184        {
185            permissionManager.addPermission(permission2);
186        }
187        catch (EntityExistsException uee)
188        {
189            // good
190        }
191    }
192}
