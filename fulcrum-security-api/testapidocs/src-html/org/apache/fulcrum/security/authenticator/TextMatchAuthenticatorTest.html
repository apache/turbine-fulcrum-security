037public class TextMatchAuthenticatorTest 
038{
039        @Test
040    public void testAuthenticate() throws Exception
041    {
042        User user = new DynamicUserImpl();
043        user.setName("Bob");
044        user.setPassword("myPassword");
045        Authenticator authenticator = new TextMatchAuthenticator();
046        assertTrue(authenticator.authenticate(user, "myPassword"));
047        assertFalse(authenticator.authenticate(user, "mypassword"));
048    }
