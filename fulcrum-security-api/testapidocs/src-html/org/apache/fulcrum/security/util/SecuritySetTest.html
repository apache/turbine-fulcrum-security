021import junit.framework.TestCase;
022
023import org.apache.fulcrum.security.entity.Group;
024import org.apache.fulcrum.security.model.dynamic.entity.impl.DynamicGroupImpl;
025
026/**
027 * Test using a SecuritySet. Useing various subclasses since it is Abstract.
028 *
029 * @author <a href="mailto:epugh@upstate.com">Eric Pugh</a>
030 * @version $Id$
031 */
032public class SecuritySetTest extends TestCase
033{
034
035    /**
036     * Defines the testcase name for JUnit.
037     *
038     * @param name
039     *            the testcase's name.
040     */
041    public SecuritySetTest(String name)
042    {
043        super(name);
044    }
045
046    public void testNull() throws Exception
047    {
048        SecuritySet<Group> securitySet = new GroupSet();
049        assertFalse(securitySet.contains(null));
050    }
051
052    public void testContainsName()
053    {
054        SecuritySet<Group> securitySet = new GroupSet();
055        assertFalse(securitySet.containsName(null));
056        Group g = new DynamicGroupImpl();
057        g.setName("BOB");
058
059        ((GroupSet) securitySet).add(g);
060        assertTrue(((GroupSet) securitySet).containsName("bob"));
061        assertTrue(((GroupSet) securitySet).containsName("BOB"));
062
063    }
064
065    public void testRemoveAll()
066    {
067        SecuritySet<Group> securitySet = new GroupSet();
068        assertFalse(securitySet.containsName(null));
069        Group g = new DynamicGroupImpl();
070        g.setName("BOB");
071        g.setId("BOB");
072
073        ((GroupSet) securitySet).add(g);
074
075        SecuritySet<Group> securitySet2 = new GroupSet();
076        assertFalse(securitySet.containsName(null));
077        g = new DynamicGroupImpl();
078        g.setName("BOB");
079        g.setId("BOB");
080
081        ((GroupSet) securitySet2).add(g);
082        securitySet.removeAll(securitySet2);
083        assertEquals(0, securitySet.size());
084    }
085
086    public void testToArray() throws Exception
087    {
088        SecuritySet<Group> securitySet = getTestData();
089        Object array[] = securitySet.toArray();
090        assertEquals(2, array.length);
091        Object array2[] = new Object[2];
092        array2[0] = "hi";
093        Object array3[] = securitySet.toArray(array2);
094        assertEquals(2, array3.length);
095    }
096
097    public void testAdd() throws Exception
098    {
099        GroupSet securitySet = (GroupSet) getTestData();
100        Group g = new DynamicGroupImpl();
101        g.setName("Michael");
102        g.setId("Michael");
103        assertTrue(securitySet.add(g));
104    }
105
106    private SecuritySet<Group> getTestData()
107    {
108        SecuritySet<Group> securitySet = new GroupSet();
109        assertFalse(securitySet.containsName(null));
110        Group g = new DynamicGroupImpl();
111        g.setName("JOE");
112        g.setId("JOE");
113
114        Group g2 = new DynamicGroupImpl();
115        g2.setName("RICK");
116        g2.setId("RICK");
117
118        ((GroupSet) securitySet).add(g);
119        ((GroupSet) securitySet).add(g2);
120
121        return securitySet;
122    }
123
124}
