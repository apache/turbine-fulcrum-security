022import java.util.Set;
023
024import org.apache.fulcrum.security.util.DataBackendException;
025
026/**
027 * Represents the "turbine" model where permissions are in a many to many
028 * relationship to roles, roles are related to groups are related to users, all
029 * in many to many relationships.
030 *
031 * @author <a href="mailto:epugh@upstate.com">Eric Pugh </a>
032 * @version $Id: TurbineRole.java 535465 2007-05-05 06:58:06Z tv $
033 */
034public interface TurbineUserGroupRoleEntity {
035
036        /**
037         * Get the User/Group/Role set associated with this entity
038         * 
039         * @param <T> TurbineUserGroupRole
040         * @return a set of User/Group/Role relations
041         * @throws DataBackendException generic exception
042         */
043        public <T extends TurbineUserGroupRole> Set<T> getUserGroupRoleSet() throws DataBackendException;
044
045        /**
046         * Set the User/Group/Role set associated with this entity
047         * 
048         * @param userGroupRoleSet a set of User/Group/Role relations
049         * @param <T> TurbineUserGroupRole
050         */
051        public <T extends TurbineUserGroupRole> void setUserGroupRoleSet(Set<T> userGroupRoleSet);
052
053        /**
054         * Add a User/Group/Role relation to this entity
055         *
056         * @param userGroupRole a User/Group/Role relation to add
057         * @throws DataBackendException generic exception
058         */
059        public void addUserGroupRole(TurbineUserGroupRole userGroupRole) throws DataBackendException;
060
061        /**
062         * Remove a User/Group/Role relation from this entity
063         *
064         * @param userGroupRole a User/Group/Role relation to remove
065         * @throws DataBackendException generic exception
066         */
067        public void removeUserGroupRole(TurbineUserGroupRole userGroupRole) throws DataBackendException;
068}
