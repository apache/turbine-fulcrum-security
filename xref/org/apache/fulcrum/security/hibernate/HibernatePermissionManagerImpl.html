31  /**
32   * This implementation persists to a database via Hibernate.
33   * 
34   * @author <a href="mailto:epugh@upstate.com">Eric Pugh</a>
35   * @version $Id: HibernatePermissionManagerImpl.java 1374014 2012-08-16
36   *          19:47:27Z tv $
37   */
38  @SuppressWarnings("unchecked")
39  public class HibernatePermissionManagerImpl extends AbstractPermissionManager
40  {
41      private PersistenceHelper persistenceHelper;
42  
43      /**
44       * Retrieves all permissions defined in the system.
45       * 
46       * @return the names of all roles defined in the system.
47       * @throws DataBackendException
48       *             if there was an error accessing the data backend.
49       */
50      public PermissionSet getAllPermissions() throws DataBackendException
51      {
52          PermissionSetsionSet.html#PermissionSet">PermissionSet permissionSet = new PermissionSet();
53          try
54          {
55              List<Permission> permissions = getPersistenceHelper().retrieveSession().createQuery("from " + Permission.class.getName()).list();
56              permissionSet.add(permissions);
57          }
58          catch (HibernateException e)
59          {
60              throw new DataBackendException("Error retrieving permission information", e);
61          }
62          return permissionSet;
63      }
64  
65      /**
66       * Renames an existing Permission.
67       * 
68       * @param permission
69       *            The object describing the permission to be renamed.
70       * @param name
71       *            the new name for the permission.
72       * @throws DataBackendException
73       *             if there was an error accessing the data backend.
74       * @throws UnknownEntityException
75       *             if the permission does not exist.
76       */
77      public synchronized void renamePermission(Permission permission, String name) throws DataBackendException, UnknownEntityException
78      {
79          boolean permissionExists = false;
80          permissionExists = checkExists(permission);
81          if (permissionExists)
82          {
83              permission.setName(name);
84              getPersistenceHelper().updateEntity(permission);
85          }
86          else
87          {
88              throw new UnknownEntityException("Unknown permission '" + permission + "'");
89          }
90      }
91  
92      /**
93       * Determines if the <code>Permission</code> exists in the security system.
94       * 
95       * @param permissionName
96       *            a <code>Permission</code> value
97       * @return true if the permission name exists in the system, false otherwise
98       * @throws DataBackendException
99       *             when more than one Permission with the same name exists.
100      */
101     public boolean checkExists(String permissionName) throws DataBackendException
102     {
103         List<Permission> permissions;
104         try
105         {
106             permissions = getPersistenceHelper().retrieveSession()
107                     .createQuery("from " + Permission.class.getName() + " sp where sp.name=:name").setString("name", permissionName).list();
108         }
109         catch (HibernateException e)
110         {
111             throw new DataBackendException("Error retrieving permission information", e);
112         }
113         if (permissions.size() > 1)
114         {
115             throw new DataBackendException("Multiple permissions with same name '" + permissionName + "'");
116         }
117         return (permissions.size() == 1);
118     }
119 
120     /**
121      * Removes a Permission from the system.
122      * 
123      * @param permission
124      *            The object describing the permission to be removed.
125      * @throws DataBackendException
126      *             if there was an error accessing the data backend.
127      * @throws UnknownEntityException
128      *             if the permission does not exist.
129      */
130     public synchronized void removePermission(Permission permission) throws DataBackendException, UnknownEntityException
131     {
132         boolean permissionExists = false;
133         permissionExists = checkExists(permission);
134         if (permissionExists)
135         {
136             getPersistenceHelper().removeEntity(permission);
137         }
138         else
139         {
140             throw new UnknownEntityException("Unknown permission '" + permission + "'");
141         }
142     }
143 
144     /**
145      * Creates a new permission with specified attributes.
146      * 
147      * @param permission
148      *            the object describing the permission to be created.
149      * @return a new Permission object that has id set up properly.
150      * @throws DataBackendException
151      *             if there was an error accessing the data backend.
152      */
153     @Override
154     protected synchronized Permissionorg/apache/fulcrum/security/entity/Permission.html#Permission">Permission persistNewPermission(Permission permission) throws DataBackendException
155     {
156         getPersistenceHelper().addEntity(permission);
157         return permission;
158     }
159 
160     /**
161      * @return Returns the persistenceHelper.
162      */
163     public PersistenceHelper getPersistenceHelper()
164     {
165         if (persistenceHelper == null)
166         {
167             persistenceHelper = (PersistenceHelper) resolve(PersistenceHelper.ROLE);
168         }
169         return persistenceHelper;
170     }
171 
172     /**
173      * Retrieve a Permission object with specified id.
174      * 
175      * @param id
176      *            the id of the Permission.
177      * @return an object representing the Permission with specified id.
178      * @throws DataBackendException
179      *             if there was an error accessing the data backend.
180      * @throws UnknownEntityException
181      *             if the permission does not exist.
182      */
183     @Override
184     public Permission getPermissionById(Object id) throws DataBackendException, UnknownEntityException
185     {
186 
187         Permission permission = null;
188 
189         if (id != null)
190         {
191             try
192             {
193                 List<Permission> permissions = getPersistenceHelper().retrieveSession()
194                         .createQuery("from " + Permission.class.getName() + " sp where sp.id=:id").setLong("id", ((Long) id).longValue())
195                         .list();
196                 if (permissions.size() == 0)
197                 {
198                     throw new UnknownEntityException("Could not find permission by id " + id);
199                 }
200                 permission = permissions.get(0);
201 
202             }
203             catch (HibernateException e)
204             {
205                 throw new DataBackendException("Error retrieving permission information", e);
206             }
207         }
208 
209         return permission;
210     }
