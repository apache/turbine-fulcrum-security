022import static org.junit.Assert.assertTrue;
023
024import org.apache.fulcrum.security.SecurityService;
025import org.apache.fulcrum.security.model.dynamic.DynamicModelManager;
026import org.apache.fulcrum.testcontainer.BaseUnit5Test;
027import org.junit.jupiter.api.Test;
028
029/**
030 * @author <a href="mailto:marco@intermeta.de">Marco Kn&uuml;ttel</a>
031 * @version $Id: StartingSecurityServicesTest.java 535465 2007-05-05 06:58:06Z
032 *          tv $
033 */
034
035public class StartingSecurityServicesTest extends BaseUnit5Test
036{
037    private SecurityService securityService = null;
038
039
040    @Test
041    public void testStartingHibernateSecurity() throws Exception
042    {
043        this.setRoleFileName("src/test/DynamicHibernateRoleConfig.xml");
044        this.setConfigurationFileName("src/test/DynamicHibernateComponentConfig.xml");
045        securityService = (SecurityService) lookup(SecurityService.ROLE);
046        assertTrue(securityService.getUserManager() instanceof HibernateUserManagerImpl);
047        assertTrue(securityService.getRoleManager() instanceof HibernateRoleManagerImpl);
048        assertTrue(securityService.getPermissionManager() instanceof HibernatePermissionManagerImpl);
049        assertTrue(securityService.getGroupManager() instanceof HibernateGroupManagerImpl);
050        assertTrue(securityService.getModelManager() instanceof DynamicModelManager);
051    }
052
053}
