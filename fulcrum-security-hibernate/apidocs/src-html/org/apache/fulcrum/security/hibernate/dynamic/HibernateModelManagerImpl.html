048    /**
049     * Revokes a Role from a Group.
050     * 
051     * @param group
052     *            the Group.
053     * @param role
054     *            the Role.
055     * @throws DataBackendException
056     *             if there was an error accessing the data backend.
057     * @throws UnknownEntityException
058     *             if group or role is not present.
059     */
060    public synchronized void revoke(Group group, Role role) throws DataBackendException, UnknownEntityException
061    {
062        boolean groupExists = false;
063        boolean roleExists = false;
064        try
065        {
066            groupExists = getGroupManager().checkExists(group);
067            roleExists = getRoleManager().checkExists(role);
068            if (groupExists && roleExists)
069            {
070                ((DynamicGroup) group).removeRole(role);
071                ((DynamicRole) role).removeGroup(group);
072                getPersistenceHelper().updateEntity(group);
073                // updateEntity(role);
074            }
075        }
076        catch (Exception e)
077        {
078            throw new DataBackendException("revoke(Group,Role) failed", e);
079        }
080        if (!groupExists)
081        {
082            throw new UnknownEntityException("Unknown group '" + group.getName() + "'");
083        }
084        if (!roleExists)
085        {
086            throw new UnknownEntityException("Unknown role '" + role.getName() + "'");
087        }
088    }

217    /**
218     * Removes a user in a group.
219     * 
220     * This method is used when removing a user to a group
221     * 
222     * @param user
223     *            the User.
224     * @throws DataBackendException
225     *             if there was an error accessing the data backend.
226     * @throws UnknownEntityException
227     *             if the user or group is not present.
228     */
229    public synchronized void revoke(User user, Group group) throws DataBackendException, UnknownEntityException
230    {
231        boolean groupExists = false;
232        boolean userExists = false;
233        Transaction transaction = null;
234
235        try
236        {
237            groupExists = getGroupManager().checkExists(group);
238            userExists = getUserManager().checkExists(user);
239            if (groupExists && userExists)
240            {
241                Session session = getPersistenceHelper().retrieveSession();
242                transaction = session.beginTransaction();
243                ((DynamicUser) user).removeGroup(group);
244                ((DynamicGroup) group).removeUser(user);
245                session.update(user);
246                session.update(group);
247                transaction.commit();
248                transaction = null;
249            }
250        }
251        catch (DataBackendException e)
252        {
253            throw new DataBackendException("grant(Role,Permission) failed", e);
254        }
255        finally
256        {
257            if (transaction != null)
258            {
259                transaction.rollback();
260            }
261        }
262        if (!groupExists)
263        {
264            throw new UnknownEntityException("Unknown group '" + group.getName() + "'");
265        }
266        if (!userExists)
267        {
268            throw new UnknownEntityException("Unknown user '" + user.getName() + "'");
269        }
270    }

315    /**
316     * @return Returns the persistenceHelper.
317     */
318    public PersistenceHelper getPersistenceHelper()
319    {
320        if (persistenceHelper == null)
321        {
322            persistenceHelper = (PersistenceHelper) resolve(PersistenceHelper.ROLE);
323        }
324        return persistenceHelper;
325    }
