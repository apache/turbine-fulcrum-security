040/**
041 * Test user with attached object (user-role-goup relationship)
042 * 
043 * Compare difference to {@link ACLFactoryTest}.
044 * 
045 * @author <a href="mailto:gk@apache.org">Georg Kallidis</a>
046 * @version $Id$
047 */
048public class TurbineACLFactoryTest extends BaseUnit5Test
049
050{
051    protected static HsqlDB hsqlDB = null;
052    
053    protected SecurityService securityService;
054    
055    @BeforeEach
056    public void setup() throws Exception {
057        
058        hsqlDB = new HsqlDB( "src/test/fulcrum-turbine-default-schema.sql" );
059        
060        this.setRoleFileName( "src/test/TurbineTorqueRoleConfig.xml" );
061        this.setConfigurationFileName( "src/test/TurbineDefaultWithPeersComponentConfig.xml" );
062        
063        securityService = (SecurityService) lookup( SecurityService.ROLE );
064        
065    }
066    
067    @Test
068    public void testCreatingTurbineACLandModel() throws Exception
069    {
070
071        ACLFactory factory = (ACLFactory) lookup(ACLFactory.ROLE);
072        
073        TurbineModelManager modelManager  = (TurbineModelManager) lookup(TurbineModelManager.ROLE);
074        assertTrue(modelManager.getGlobalGroupName().equals("global"));
075        
076        Group global = securityService.getGroupManager().getGroupInstance("global");
077        global.setId(new Integer(1));
078        securityService.getGroupManager().addGroup(global);
079        
080        TurbineUser user = securityService.getUserManager().getUserInstance("Bob");
081        user.setId(new Integer(1));
082        securityService.getUserManager().addUser(user, "mypw");
083        
084        TurbineGroup group = securityService.getGroupManager().getGroupInstance("group1");
085        group.setId(new Integer(2));
086        securityService.getGroupManager().addGroup(group);
087        
088        TurbineRole role = securityService.getRoleManager().getRoleInstance();
089        role.setName("role1");
090        role.setId(new Integer(1));
091        securityService.getRoleManager().addRole( role );
092        
093        TurbinePermission permission = securityService.getPermissionManager().getPermissionInstance();
094        permission.setName("permission1");
095        permission.setId(new Integer(1)); 
096        securityService.getPermissionManager().addPermission(permission);
097
098        role.addPermission(permission);
099        
100        // need to save it as TurbineAccessControlListImpl refreshes role permission set! 
101        securityService.<TurbineModelManager> getModelManager().grant(role, permission);
102        
103        TurbineUserGroupRole ugr = new TurbineUserGroupRole();
104        ugr.setGroup(group);
105        ugr.setRole(role);
106        ugr.setUser(user);
107        user.addUserGroupRole(ugr);
108        
109        securityService.<TurbineModelManager> getModelManager().grant( user, group, role );
110        
111        //securityService.getUserManager().saveUser(user);
112        
113        AccessControlList acl = factory.getAccessControlList(user);
114        
115        assertTrue(acl instanceof TurbineAccessControlList);
116        
117        TurbineAccessControlList tacl = (TurbineAccessControlList) acl;
118        
119        assertTrue(tacl.hasPermission(permission, group));
120    }
121
122}
