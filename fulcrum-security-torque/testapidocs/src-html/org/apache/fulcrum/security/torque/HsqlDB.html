022import java.io.File;
023import java.io.FileInputStream;
024import java.io.InputStream;
025import java.sql.Connection;
026import java.sql.DriverManager;
027import java.sql.Statement;
028import java.util.ArrayList;
029import java.util.List;
030import java.util.Scanner;
031
032import org.apache.commons.lang3.StringUtils;
033import org.hsqldb.jdbcDriver;
034
035/**
036 * This Class provides a nice Setup for a Hypersonic SQL Server, which will only
037 * be held in Memory.
038 *
039 * It allows loading of .sql Files, to setup Tests easily.
040 *
041 * @author <a href="jh@byteaction.de">J&#252;rgen Hoffmann</a>
042 */
043public class HsqlDB {
044        private Connection connection = null;
045        private static String URI = "jdbc:hsqldb:.";
046
047        public HsqlDB(String loadFile) throws Exception {
048                Class.forName(jdbcDriver.class.getName());
049                this.connection = DriverManager.getConnection(URI, "SA", "");
050                if (StringUtils.isNotEmpty(loadFile)) {
051                        loadSqlFile(loadFile);
052                }
053        }
054
055        public HsqlDB(File loadFile) throws Exception {
056                this(loadFile.getAbsolutePath());
057        }
058
059        public Connection getConnection() {
060                return connection;
061        }
062
063        public void close() {
064                try {
065                        connection.close();
066                } catch (Exception e) {
067                }
068        }
069
070        /**
071         * Load a sql file and process its statements
072         */
073        private void loadSqlFile(String fileName) throws Exception {
074                Statement statement = null;
075                try {
076                        statement = connection.createStatement();
077                        // execute each statement on its own
078                        List<String> commands = getFileContents(fileName);
079                        for (String cmd : commands)
080                                statement.executeUpdate(cmd);
081
082                } finally {
083                        if (statement != null) {
084                                statement.close();
085                        }
086                }
087        }
088
089        private List<String> getFileContents(String fileName) throws Exception {
090
091                StringBuilder sb = new StringBuilder(1000);
092                Scanner s = null;
093                try {
094                        InputStream is = new FileInputStream(new File(fileName));
095                        s = new Scanner(is);
096                        s.useDelimiter("(\r?\n)");
097
098                        boolean inComment = false;
099                        while (s.hasNext()) {
100                                String line = s.next();
101
102                                if (inComment == true) {
103                                        if (line.endsWith("*/"))
104                                                inComment = false;
105                                } else {
106                                        // test for comments
107                                        if (!line.startsWith("--")) {
108                                                if (line.trim().length() > 0) {
109                                                        sb.append(line);
110                                                }
111                                        }
112                                }
113                        }
114                } finally {
115                        if (s != null)
116                                s.close();
117                }
118
119                String sql = sb.toString();
120                sb = new StringBuilder();
121                String[] commands = sql.split(";");
122                List<String> sqlCmds = new ArrayList<>();
123                for (String cmd : commands)
124                        sqlCmds.add(cmd + ";\n");
125
126                return sqlCmds;
127        }
128
129        public void addSQL(String sqlFile) throws Exception {
130                if (StringUtils.isNotEmpty(sqlFile)) {
131                        loadSqlFile(sqlFile);
132                }
133        }
134
135        public void addSQL(File sqlFile) throws Exception {
136                this.addSQL(sqlFile.getAbsolutePath());
137        }
138}
