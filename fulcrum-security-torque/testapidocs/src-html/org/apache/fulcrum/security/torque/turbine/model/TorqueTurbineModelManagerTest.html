022import static org.junit.jupiter.api.Assertions.fail;
023
024import org.apache.fulcrum.security.SecurityService;

027import org.apache.fulcrum.security.torque.om.TorqueTurbineGroupPeer;
028import org.apache.fulcrum.security.torque.om.TorqueTurbinePermissionPeer;
029import org.apache.fulcrum.security.torque.om.TorqueTurbineRolePeer;
030import org.apache.fulcrum.security.torque.om.TorqueTurbineRolePermissionPeer;
031import org.apache.fulcrum.security.torque.om.TorqueTurbineUserGroupRolePeer;
032import org.apache.fulcrum.security.torque.om.TorqueTurbineUserPeer;
033import org.apache.torque.TorqueException;
034import org.apache.torque.criteria.Criteria;
035import org.junit.jupiter.api.AfterEach;
036import org.junit.jupiter.api.BeforeEach;
037
038/**
039 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
040 * @author <a href="jh@byteaction.de">J&#252;rgen Hoffmann</a>
041 * @version $Id:$
042 */
043public class TorqueTurbineModelManagerTest
044    extends AbstractTurbineModelManagerTest
045{
046    protected static HsqlDB hsqlDB = null;
047
048        @BeforeEach
049        public void setUp() throws Exception
050    {
051        try
052        {
053            hsqlDB = new HsqlDB("src/test/fulcrum-turbine-schema.sql");
054            hsqlDB.addSQL("src/test/id-table-schema.sql");
055            hsqlDB.addSQL("src/test/fulcrum-turbine-schema-idtable-init.sql");
056
057            this.setRoleFileName("src/test/TurbineTorqueRoleConfig.xml");
058            this.setConfigurationFileName("src/test/TurbineTorqueComponentConfig.xml");
059            securityService = (SecurityService) lookup(SecurityService.ROLE);
060            super.setUp();
061        }
062        catch (Exception e)
063        {
064                fail(e.toString());
065        }
066
067    }
068
069        @AfterEach
070        public void tearDown()
071    {
072        // cleanup tables
073        try
074        {
075            Criteria criteria = new Criteria();
076            criteria.where(TorqueTurbineUserGroupRolePeer.USER_ID, -1, Criteria.GREATER_THAN);
077            TorqueTurbineUserGroupRolePeer.doDelete(criteria);
078
079            criteria = new Criteria();
080            criteria.where(TorqueTurbineRolePermissionPeer.ROLE_ID, 0, Criteria.GREATER_THAN);
081            TorqueTurbineRolePermissionPeer.doDelete(criteria);
082
083            criteria = new Criteria();
084            criteria.where(TorqueTurbineUserPeer.USER_ID, 0, Criteria.GREATER_THAN);
085            TorqueTurbineUserPeer.doDelete(criteria);
086
087            criteria = new Criteria();
088            criteria.where(TorqueTurbineGroupPeer.GROUP_ID, 0, Criteria.GREATER_THAN);
089            TorqueTurbineGroupPeer.doDelete(criteria);
090
091            criteria = new Criteria();
092            criteria.where(TorqueTurbineRolePeer.ROLE_ID, 0, Criteria.GREATER_THAN);
093            TorqueTurbineRolePeer.doDelete(criteria);
094
095            criteria = new Criteria();
096            criteria.where(TorqueTurbinePermissionPeer.PERMISSION_ID, 0, Criteria.GREATER_THAN);
097            TorqueTurbinePermissionPeer.doDelete(criteria);
098        }
099        catch (TorqueException e)
100        {
101                fail(e.toString());
102        }
103
104        modelManager = null;
105        securityService = null;
106    }
107
108}
