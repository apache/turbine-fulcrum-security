034import org.apache.fulcrum.security.torque.om.TorqueDynamicUserPeer;

041/**
042 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
043 * @author <a href="jh@byteaction.de">J&#252;rgen Hoffmann</a>
044 * @version $Id:$
045 */
046public class TorqueDynamicModelManagerTest extends AbstractDynamicModelManagerTest
047{
048    protected static HsqlDB hsqlDB = null;
049
050    @Override
051        @BeforeEach
052    public void setUp()
053    {
054        try
055        {
056            hsqlDB = new HsqlDB("src/test/fulcrum-dynamic-schema.sql");
057            hsqlDB.addSQL("src/test/id-table-schema.sql");
058            hsqlDB.addSQL("src/test/fulcrum-dynamic-schema-idtable-init.sql");
059
060            this.setRoleFileName("src/test/DynamicTorqueRoleConfig.xml");
061            this.setConfigurationFileName("src/test/DynamicTorqueComponentConfig.xml");
062            securityService = (SecurityService) lookup(SecurityService.ROLE);
063            super.setUp();
064        }
065        catch (Exception e)
066        {
067            fail(e.toString());
068        }
069    }
070
071    @Override
072    @AfterEach
073        public void tearDown()
074    {
075        // cleanup tables
076        try
077        {
078            Criteria criteria = new Criteria();
079            criteria.where(TorqueDynamicUserGroupPeer.USER_ID, 0, Criteria.GREATER_THAN);
080            TorqueDynamicUserGroupPeer.doDelete(criteria);
081
082            criteria = new Criteria();
083            criteria.where(TorqueDynamicGroupRolePeer.GROUP_ID, 0, Criteria.GREATER_THAN);
084            TorqueDynamicGroupRolePeer.doDelete(criteria);
085
086            criteria = new Criteria();
087            criteria.where(TorqueDynamicRolePermissionPeer.ROLE_ID, 0, Criteria.GREATER_THAN);
088            TorqueDynamicRolePermissionPeer.doDelete(criteria);
089
090            criteria = new Criteria();
091            criteria.where(TorqueDynamicUserDelegatesPeer.DELEGATEE_USER_ID, 0, Criteria.GREATER_THAN);
092            TorqueDynamicUserDelegatesPeer.doDelete(criteria);
093
094            criteria = new Criteria();
095            criteria.where(TorqueDynamicUserPeer.USER_ID, 0, Criteria.GREATER_THAN);
096            TorqueDynamicUserPeer.doDelete(criteria);
097
098            criteria = new Criteria();
099            criteria.where(TorqueDynamicGroupPeer.GROUP_ID, 0, Criteria.GREATER_THAN);
100            TorqueDynamicGroupPeer.doDelete(criteria);
101
102            criteria = new Criteria();
103            criteria.where(TorqueDynamicRolePeer.ROLE_ID, 0, Criteria.GREATER_THAN);
104            TorqueDynamicRolePeer.doDelete(criteria);
105
106            criteria = new Criteria();
107            criteria.where(TorqueDynamicPermissionPeer.PERMISSION_ID, 0, Criteria.GREATER_THAN);
108            TorqueDynamicPermissionPeer.doDelete(criteria);
109        }
110        catch (TorqueException e)
111        {
112            fail(e.toString());
113        }
114
115        securityService = null;
116    }
117
118}
