785        else
786        {
787            // the following code is to determine if a new query is
788            // called for.  If the criteria is the same as the last
789            // one, just return the collection.
790            criteria.and(org.apache.fulcrum.security.torque.om.TurbineUserGroupRolePeer.USER_ID, getEntityId());
791            if (lastTurbineUserGroupRoleCriteria == null
792                        || !lastTurbineUserGroupRoleCriteria.equals(criteria))
793            {
794                collTurbineUserGroupRoles = org.apache.fulcrum.security.torque.om.TurbineUserGroupRolePeer.doSelectJoinTurbineRole(criteria, connection);
795                for (TurbineUserGroupRole turbineUserGroupRole : collTurbineUserGroupRoles)
796                {
797                    turbineUserGroupRole.setTurbineUser((TurbineUser) this);
798                }
799            }
800        }
801        lastTurbineUserGroupRoleCriteria = criteria;
802
803        return collTurbineUserGroupRoles;
804    }

843     * is meant to be used as part of a transaction, otherwise use
844     * the save() method and the connection details will be handled
845     * internally.
846     *
847     * @param con the connection to use for saving the object, not null.
848     *
849     * @throws TorqueException if an error occurs during saving.
850     */
851    public void save(Connection con) 
852            throws TorqueException
853    {
854        if (isSaving())
855        {
856            return;
857        }
858        try
859        {
860            setSaving(true);
861            // If this object has been modified, then save it to the database.
862            if (isModified())
863            {
864                if (isNew())
865                {
866                    TurbineUserPeer.doInsert((TurbineUser) this, con);
867                    setNew(false);
868                }
869                else
870                {
871                    TurbineUserPeer.doUpdate((TurbineUser) this, con);
872                }
873            }
874
875            if (isTurbineUserGroupRolesInitialized())
876            {
877                for (TurbineUserGroupRole collTurbineUserGroupRoles : getTurbineUserGroupRoles())
878                {
879                    collTurbineUserGroupRoles.save(con);

881            }
882        }
883        finally
884        {
885            setSaving(false);
